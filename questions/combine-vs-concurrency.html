Overview: Two Approaches to Async Programming

Combine (iOS 13+) and Swift Concurrency (iOS 15+) are both powerful tools for handling asynchronous operations in SwiftUI, but they use fundamentally different paradigms and approaches.

import Combine
import SwiftUI

class WeatherViewModel: ObservableObject {
    @Published var weather: Weather?
    @Published var isLoading = false
    @Published var errorMessage: String?
    
    private var cancellables = Set()
    private let weatherService = WeatherService()
    
    func loadWeather(for city: String) {
        isLoading = true
        errorMessage = nil
        
        weatherService.fetchWeather(for: city)
            .receive(on: DispatchQueue.main)
            .sink(
                receiveCompletion: { [weak self] completion in
                    self?.isLoading = false
                    if case .failure(let error) = completion {
                        self?.errorMessage = error.localizedDescription
                    }
                },
                receiveValue: { [weak self] weather in
                    self?.weather = weather
                }
            )
            .store(in: &cancellables)
    }
    
    func searchWeather(searchText: AnyPublisher) {
        searchText
            .debounce(for: .milliseconds(500), scheduler: DispatchQueue.main)
            .removeDuplicates()
            .filter { !$0.isEmpty }
            .flatMap { city in
                self.weatherService.fetchWeather(for: city)
                    .catch { _ in Just(Weather.placeholder) }
            }
            .receive(on: DispatchQueue.main)
            .assign(to: \.weather, on: self)
            .store(in: &cancellables)
    }
}

struct WeatherService {
    func fetchWeather(for city: String) -> AnyPublisher {
        let url = URL(string: "https://api.weather.com/\(city)")!
        
        return URLSession.shared.dataTaskPublisher(for: url)
            .map(\.data)
            .decode(type: Weather.self, decoder: JSONDecoder())
            .eraseToAnyPublisher()
    }
}

import SwiftUI

@MainActor
class WeatherViewModel: ObservableObject {
    @Published var weather: Weather?
    @Published var isLoading = false
    @Published var errorMessage: String?
    
    private let weatherService = WeatherService()
    private var searchTask: Task?
    
    func loadWeather(for city: String) async {
        isLoading = true
        errorMessage = nil
        
        do {
            let weather = try await weatherService.fetchWeather(for: city)
            self.weather = weather
        } catch {
            self.errorMessage = error.localizedDescription
        }
        
        isLoading = false
    }
    
    func searchWeather(for city: String) {
        // Cancel previous search
        searchTask?.cancel()
        
        searchTask = Task {
            // Debounce
            try? await Task.sleep(nanoseconds: 500_000_000)
            
            guard !Task.isCancelled, !city.isEmpty else { return }
            
            do {
                let weather = try await weatherService.fetchWeather(for: city)
                self.weather = weather
            } catch {
                if !Task.isCancelled {
                    self.errorMessage = error.localizedDescription
                }
            }
        }
    }
}

struct WeatherService {
    func fetchWeather(for city: String) async throws -> Weather {
        let url = URL(string: "https://api.weather.com/\(city)")!
        let (data, _) = try await URLSession.shared.data(from: url)
        return try JSONDecoder().decode(Weather.self, from: data)
    }
}

When to Use Each:

ğŸ”„ Use Combine When:
  • Complex Data Streams: Multiple publishers, transformations
  • Reactive UI: Real-time data updates, search-as-you-type
  • Timer-based Operations: Periodic updates, debouncing
  • Legacy Support: Need to support iOS 13-14
  • Data Pipeline: Complex data transformation chains
âš¡ Use Swift Concurrency When:
  • Simple Async Operations: API calls, file operations
  • Sequential Operations: Step-by-step async workflows
  • Better Readability: Linear, easy-to-follow code
  • Modern Codebase: iOS 15+ target deployment
  • Error Handling: Standard try/catch patterns

Performance Comparison:

Metric Combine Swift Concurrency
Memory Usage Higher (publisher chains) Lower (structured concurrency)
CPU Overhead Moderate (operator chains) Lower (native async support)
Debugging Complex (operator chains) Simpler (linear flow)
Compilation Time Slower (complex types) Faster (native support)

Best Practices:

Combine Best Practices
  • Always store cancellables properly
  • Use weak self in sink closures
  • Handle errors explicitly in publisher chains
  • Use appropriate schedulers for UI updates
Swift Concurrency Best Practices
  • Use @MainActor for UI-related async functions
  • Handle Task cancellation properly
  • Use structured concurrency (async let, TaskGroup)
  • Avoid blocking the main actor
