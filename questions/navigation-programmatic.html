How do you manage navigation programmatically in SwiftUI?

🎯 الفكرة الأساسية

Programmatic Navigation في SwiftUI بيتم من خلال controlling navigation state بطرق مختلفة حسب نوع التطبيق والـ iOS version. دي الطرق الحديثة والفعالة لإدارة Navigation.

📝 الشرح بالعربي

Navigation في SwiftUI اتطور كتير، خاصة مع iOS 16 وما بعده. دلوقتي عندنا طرق مختلفة لإدارة التنقل programmatically، من الـ NavigationStack الجديد لحد الطرق التقليدية بالـ NavigationView.

🚀 iOS 16+ Modern Approach: NavigationStack

1. NavigationStack with NavigationPath

الطريقة الأحدث والأفضل لإدارة Navigation باستخدام NavigationPath


struct ContentView: View {
    @State private var navigationPath = NavigationPath()
    
    var body: some View {
        NavigationStack(path: $navigationPath) {
            VStack {
                Button("Go to Profile") {
                    navigationPath.append("profile")
                }
                
                Button("Go to Settings") {
                    navigationPath.append("settings")
                }
                
                Button("Go to User Detail") {
                    navigationPath.append(User(id: 1, name: "Ahmed"))
                }
            }
            .navigationDestination(for: String.self) { value in
                switch value {
                case "profile":
                    ProfileView()
                case "settings":
                    SettingsView()
                default:
                    Text("Unknown destination")
                }
            }
            .navigationDestination(for: User.self) { user in
                UserDetailView(user: user)
            }
            .navigationTitle("Home")
        }
    }
}

struct User: Hashable {
    let id: Int
    let name: String
}
2. Controlling Navigation Stack

يمكنك التحكم الكامل في navigation stack من خلال NavigationPath


class NavigationManager: ObservableObject {
    @Published var navigationPath = NavigationPath()
    
    func navigateToProfile() {
        navigationPath.append("profile")
    }
    
    func navigateToUser(_ user: User) {
        navigationPath.append(user)
    }
    
    func goBack() {
        navigationPath.removeLast()
    }
    
    func goToRoot() {
        navigationPath.removeLast(navigationPath.count)
    }
    
    func navigateToSettingsFromAnywhere() {
        // Clear stack and go to settings
        navigationPath = NavigationPath()
        navigationPath.append("settings")
    }
}

struct ContentView: View {
    @StateObject private var navManager = NavigationManager()
    
    var body: some View {
        NavigationStack(path: $navManager.navigationPath) {
            VStack {
                Button("Navigate to Profile") {
                    navManager.navigateToProfile()
                }
                
                Button("Go to Root") {
                    navManager.goToRoot()
                }
            }
            .navigationDestination(for: String.self) { destination in
                DestinationView(destination: destination, navManager: navManager)
            }
        }
    }
}

📱 Pre-iOS 16: NavigationView Approach

للتطبيقات اللي بتدعم iOS 15 وأقل

استخدم NavigationView مع @State variables للـ navigation control


struct LegacyNavigationView: View {
    @State private var showProfile = false
    @State private var showSettings = false
    @State private var selectedUser: User?
    
    var body: some View {
        NavigationView {
            VStack {
                Button("Show Profile") {
                    showProfile = true
                }
                
                Button("Show Settings") {
                    showSettings = true
                }
                
                Button("Show User Detail") {
                    selectedUser = User(id: 1, name: "Ahmed")
                }
            }
            .navigationTitle("Home")
            .navigationBarTitleDisplayMode(.large)
            .background(
                VStack {
                    NavigationLink(
                        destination: ProfileView(),
                        isActive: $showProfile
                    ) { EmptyView() }
                    
                    NavigationLink(
                        destination: SettingsView(),
                        isActive: $showSettings
                    ) { EmptyView() }
                    
                    NavigationLink(
                        destination: UserDetailView(user: selectedUser ?? User(id: 0, name: "")),
                        isActive: .constant(selectedUser != nil)
                    ) { EmptyView() }
                        .onDisappear {
                            selectedUser = nil
                        }
                }
                .hidden()
            )
        }
    }
}

🔄 Deep Linking & URL-based Navigation

Handling Deep Links

التعامل مع Deep Links وتوجيه المستخدم للصفحة المطلوبة


@main
struct MyApp: App {
    var body: some Scene {
        WindowGroup {
            ContentView()
                .onOpenURL { url in
                    handleDeepLink(url)
                }
        }
    }
    
    private func handleDeepLink(_ url: URL) {
        // Parse URL and navigate accordingly
        // myapp://profile -> navigate to profile
        // myapp://user/123 -> navigate to user with ID 123
        
        let components = URLComponents(url: url, resolvingAgainstBaseURL: true)
        
        switch components?.host {
        case "profile":
            NavigationManager.shared.navigateToProfile()
        case "user":
            if let userID = components?.path.components(separatedBy: "/").last,
               let id = Int(userID) {
                NavigationManager.shared.navigateToUser(User(id: id, name: "User \(id)"))
            }
        default:
            break
        }
    }
}

🎛️ Advanced Navigation Patterns

1. Tab-based Navigation with Programmatic Control

struct TabNavigationView: View {
    @State private var selectedTab = 0
    @State private var homeNavigationPath = NavigationPath()
    @State private var profileNavigationPath = NavigationPath()
    
    var body: some View {
        TabView(selection: $selectedTab) {
            NavigationStack(path: $homeNavigationPath) {
                HomeView()
                    .navigationDestination(for: String.self) { destination in
                        DestinationView(destination: destination)
                    }
            }
            .tabItem {
                Image(systemName: "house")
                Text("Home")
            }
            .tag(0)
            
            NavigationStack(path: $profileNavigationPath) {
                ProfileView()
                    .navigationDestination(for: String.self) { destination in
                        DestinationView(destination: destination)
                    }
            }
            .tabItem {
                Image(systemName: "person")
                Text("Profile")
            }
            .tag(1)
        }
    }
    
    func navigateToProfileTab() {
        selectedTab = 1 // Switch to profile tab
        profileNavigationPath.append("settings") // Navigate within profile tab
    }
}
2. Modal Presentation with Navigation

struct ModalNavigationExample: View {
    @State private var showModal = false
    @State private var modalNavigationPath = NavigationPath()
    
    var body: some View {
        VStack {
            Button("Show Modal") {
                showModal = true
            }
        }
        .sheet(isPresented: $showModal) {
            NavigationStack(path: $modalNavigationPath) {
                ModalRootView()
                    .navigationDestination(for: String.self) { destination in
                        ModalDestinationView(destination: destination)
                    }
                    .navigationBarTitleDisplayMode(.inline)
                    .toolbar {
                        ToolbarItem(placement: .navigationBarLeading) {
                            Button("Cancel") {
                                showModal = false
                            }
                        }
                        ToolbarItem(placement: .navigationBarTrailing) {
                            Button("Done") {
                                // Process and dismiss
                                showModal = false
                            }
                        }
                    }
            }
        }
    }
}

📱 Notification-based Navigation

Navigation من خلال Push Notifications

class NotificationNavigationHandler: ObservableObject {
    @Published var shouldNavigateToChat = false
    @Published var shouldNavigateToProfile = false
    @Published var chatID: String?
    
    func handleNotification(_ userInfo: [AnyHashable: Any]) {
        guard let type = userInfo["type"] as? String else { return }
        
        switch type {
        case "chat":
            chatID = userInfo["chatID"] as? String
            shouldNavigateToChat = true
        case "profile":
            shouldNavigateToProfile = true
        default:
            break
        }
    }
}

struct ContentView: View {
    @StateObject private var navigationHandler = NotificationNavigationHandler()
    @State private var navigationPath = NavigationPath()
    
    var body: some View {
        NavigationStack(path: $navigationPath) {
            HomeView()
                .navigationDestination(for: String.self) { destination in
                    DestinationView(destination: destination)
                }
        }
        .onReceive(NotificationCenter.default.publisher(for: UIApplication.didBecomeActiveNotification)) { _ in
            // Handle navigation when app becomes active from notification
            if navigationHandler.shouldNavigateToChat {
                if let chatID = navigationHandler.chatID {
                    navigationPath.append("chat/\(chatID)")
                }
                navigationHandler.shouldNavigateToChat = false
            }
        }
    }
}

🎯 Best Practices

نصائح هامة لـ Programmatic Navigation:

  1. استخدم NavigationStack: للـ iOS 16+ projects
  2. Centralized Navigation: استخدم NavigationManager class
  3. Type Safety: استخدم Hashable types بدل Strings لما أمكن
  4. Deep Linking: اتعامل مع URLs بشكل صحيح
  5. State Management: خلي navigation state منفصل عن UI state
  6. Memory Management: تأكد من عدم retain cycles في navigation handlers

📊 الخلاصة

Programmatic Navigation في SwiftUI:

  • استخدم NavigationStack + NavigationPath للـ iOS 16+
  • للـ legacy support، استخدم NavigationView + @State
  • اعمل centralized navigation manager للتحكم
  • اتعامل مع deep links و push notifications
  • استخدم type-safe navigation لما أمكن
