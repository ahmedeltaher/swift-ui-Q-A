Combine vs Swift Concurrency in SwiftUI

Answer in English

Overview of Both Approaches

Combine is a reactive programming framework that handles asynchronous events over time, while Swift Concurrency (async/await) provides a more imperative approach to asynchronous programming. Both can be used effectively in SwiftUI for data flow management.

Key Differences:

Aspect Combine Swift Concurrency
Programming Style Reactive/Declarative Imperative/Sequential
Learning Curve Steeper (reactive concepts) More intuitive (looks like sync code)
Error Handling Through operators (.catch, .retry) Traditional try-catch blocks
Cancellation Automatic with AnyCancellable Manual with Task.cancel()
Stream Processing Excellent (built for streams) Good (with AsyncSequence)
SwiftUI Integration Native (@Published, onReceive) Good (.task modifier)

import Combine
import SwiftUI

class UserViewModel: ObservableObject {
    @Published var users: [User] = []
    @Published var isLoading = false
    @Published var errorMessage: String?
    
    private var cancellables = Set()
    private let apiService = APIService()
    
    func loadUsers() {
        isLoading = true
        errorMessage = nil
        
        apiService.fetchUsers()
            .receive(on: DispatchQueue.main)
            .sink(
                receiveCompletion: { [weak self] completion in
                    self?.isLoading = false
                    if case .failure(let error) = completion {
                        self?.errorMessage = error.localizedDescription
                    }
                },
                receiveValue: { [weak self] users in
                    self?.users = users
                }
            )
            .store(in: &cancellables)
    }
    
    func searchUsers(query: String) {
        // Debounced search with Combine
        Just(query)
            .debounce(for: .milliseconds(300), scheduler: DispatchQueue.main)
            .removeDuplicates()
            .flatMap { query -> AnyPublisher<[User], Never> in
                guard !query.isEmpty else {
                    return Just([]).eraseToAnyPublisher()
                }
                
                return self.apiService.searchUsers(query: query)
                    .catch { _ in Just([]) }
                    .eraseToAnyPublisher()
            }
            .receive(on: DispatchQueue.main)
            .sink { [weak self] users in
                self?.users = users
            }
            .store(in: &cancellables)
    }
}

struct UsersView: View {
    @StateObject private var viewModel = UserViewModel()
    @State private var searchQuery = ""
    
    var body: some View {
        NavigationView {
            VStack {
                SearchBar(text: $searchQuery)
                    .onReceive(viewModel.$users) { users in
                        // React to user changes
                        print("Users updated: \(users.count)")
                    }
                
                if viewModel.isLoading {
                    ProgressView("Loading users...")
                } else if let error = viewModel.errorMessage {

                        .foregroundColor(.red)
                } else {
                    List(viewModel.users, id: \.id) { user in
                        UserRowView(user: user)
                    }
                }
            }
        }
        .onAppear {
            viewModel.loadUsers()
        }
        .onChange(of: searchQuery) { query in
            viewModel.searchUsers(query: query)
        }
    }

import SwiftUI

@MainActor
class UserViewModel: ObservableObject {
    @Published var users: [User] = []
    @Published var isLoading = false
    @Published var errorMessage: String?
    
    private let apiService = APIService()
    private var searchTask: Task?
    
    func loadUsers() async {
        isLoading = true
        errorMessage = nil
        
        do {
            let fetchedUsers = try await apiService.fetchUsers()
            users = fetchedUsers
        } catch {
            errorMessage = error.localizedDescription
        }
        
        isLoading = false
    }
    
    func searchUsers(query: String) {
        // Cancel previous search
        searchTask?.cancel()
        
        searchTask = Task {
            // Debounce manually
            try? await Task.sleep(nanoseconds: 300_000_000)
            
            guard !Task.isCancelled else { return }
            guard !query.isEmpty else {
                users = []
                return
            }
            
            do {
                let searchResults = try await apiService.searchUsers(query: query)
                guard !Task.isCancelled else { return }
                users = searchResults
            } catch {
                guard !Task.isCancelled else { return }
                errorMessage = error.localizedDescription
            }
        }
    }
}

struct UsersView: View {
    @StateObject private var viewModel = UserViewModel()
    @State private var searchQuery = ""
    
    var body: some View {
        NavigationView {
            VStack {
                SearchBar(text: $searchQuery)
                
                if viewModel.isLoading {
                    ProgressView("Loading users...")
                } else if let error = viewModel.errorMessage {
                    Text("Error: \(error)")
                        .foregroundColor(.red)
                } else {
                    List(viewModel.users, id: \.id) { user in
                        UserRowView(user: user)
                    }
                }
            }
        }
        .task {
            await viewModel.loadUsers()

Advanced Use Cases:

1. Real-time Data Updates
Combine - WebSocket Stream
class RealtimeDataManager: ObservableObject {
    @Published var messages: [Message] = []
    private var cancellables = Set()
    
    func connectToWebSocket() {
        WebSocketService.shared.messagePublisher
            .receive(on: DispatchQueue.main)
            .sink { [weak self] message in
                self?.messages.append(message)
            }
            .store(in: &cancellables)
    }
}
Swift Concurrency - AsyncSequence
@MainActor
class RealtimeDataManager: ObservableObject {
    @Published var messages: [Message] = []
    private var connectionTask: Task?
    
    func connectToWebSocket() {
        connectionTask = Task {
            for await message in WebSocketService.shared.messages {
                guard !Task.isCancelled else { break }
                messages.append(message)
            }
        }
    }
}
2. Complex Data Transformations
Combine - Operators Chain
func processUserData() {
    apiService.fetchUsers()
        .map { users in
            users.filter { $0.isActive }
        }
        .flatMap { activeUsers in
            Publishers.Sequence(sequence: activeUsers)
                .flatMap { user in
                    self.apiService.fetchUserDetails(id: user.id)
                }
                .collect()
        }
        .receive(on: DispatchQueue.main)
        .sink(
            receiveCompletion: { completion in
                // Handle completion
            },
            receiveValue: { [weak self] userDetails in
                self?.userDetails = userDetails
            }
        )
        .store(in: &cancellables)
}
Swift Concurrency - Structured Concurrency
func processUserData() async {
    do {
        let users = try await apiService.fetchUsers()
        let activeUsers = users.filter { $0.isActive }
        
        let userDetails = try await withThrowingTaskGroup(of: UserDetail.self) { group in
            for user in activeUsers {
                group.addTask {
                    try await self.apiService.fetchUserDetails(id: user.id)
                }
            }
            
            var results: [UserDetail] = []
            for try await detail in group {
                results.append(detail)
            }
            return results
        }
        
        await MainActor.run {
            self.userDetails = userDetails
        }
    } catch {
        await MainActor.run {
            self.errorMessage = error.localizedDescription
        }
    }
}

SwiftUI Integration Patterns:

Combine Integration
  • @Published: Automatic UI updates
  • .onReceive(): React to publisher values
  • AnyCancellable: Automatic subscription management
  • Operators: Rich data transformation pipeline
Swift Concurrency Integration
  • .task: Lifecycle-aware async operations
  • @MainActor: Thread-safe UI updates
  • Task cancellation: Automatic cleanup on view disappear
  • Structured concurrency: Clear async operation hierarchy

Performance Considerations:

Scenario Combine Swift Concurrency
Memory Usage Higher (operators overhead) Lower (minimal overhead)
CPU Usage Can be intensive with complex chains Generally more efficient
Debugging Complex operator chains Easier stack traces
Testing Requires Combine test utilities Standard async testing

When to Choose Which:

Choose Combine When:
  • Working with reactive streams of data
  • Need complex data transformations
  • Building reactive UI patterns
  • Integrating with Combine-based libraries
  • Need built-in operators (debounce, throttle, etc.)
Choose Swift Concurrency When:
  • Simple async operations
  • Better performance is critical
  • Team prefers imperative style
  • Working with async APIs
  • iOS 15+ minimum deployment target

Hybrid Approach:

// You can combine both approaches
class HybridViewModel: ObservableObject {
    @Published var data: [Item] = []
    @Published var isLoading = false
    
    private var cancellables = Set()
    
    func loadData() {
        isLoading = true
        
        // Use async/await for simple operations
        Task {
            do {
                let items = try await APIService.fetchItems()
                await MainActor.run {
                    self.data = items
                    self.isLoading = false
                }
            } catch {
                await MainActor.run {
                    self.isLoading = false
                }
            }
        }
    }
    
    func setupRealtimeUpdates() {
        // Use Combine for reactive streams
        NotificationCenter.default
            .publisher(for: .dataUpdated)
            .compactMap { $0.object as? [Item] }
            .receive(on: DispatchQueue.main)
            .sink { [weak self] items in
                self?.data = items
            }
            .store(in: &cancellables)
    }
}
